       export function FocusProvider({ children }) {
         const [focusedArea, setFocusedArea] = useState('global');
         return (
           <FocusContext.Provider value={{ focusedArea, setFocusedArea }}>
             {children}
           </FocusContext.Provider>
         );
       }

         const context = useContext(FocusContext);
         if (!context) throw new Error('Must be used within FocusProvider');
         return context;
       }

    2. Define shortcuts configuration:
       - Global shortcuts (always active):
         * Ctrl+O: Open file (trigger file open dialog)
         * Ctrl+S: Save file (trigger save operation)
         * Ctrl+Shift+F: Find in canvas (open canvas search)
         * F1 or Ctrl+?: Show keyboard shortcuts help
       
       - Editor-specific shortcuts (when NOT in editor):
         * None - let Monaco handle all editor shortcuts
       
       - Canvas-specific shortcuts:
         * Ctrl++ / Ctrl+-: Zoom in/out
         * Ctrl+0: Reset zoom
         * Delete: Delete selected node
    
    3. Implementation with context awareness:
       ```typescript
       export function useKeyboardShortcuts() {
         const { focusedArea } = useFocusContext();
         const { openFile, saveFile } = useFileStore(); // or appropriate store
         const { openSearch } = useSearchStore();
         
         // Global shortcuts - always active
         useHotkeys('ctrl+o', (e) => {
           e.preventDefault();
           openFile();
         }, { enableOnFormTags: true });
         
         useHotkeys('ctrl+s', (e) => {
           e.preventDefault();
           saveFile();
         }, { enableOnFormTags: true });
         
         useHotkeys('ctrl+shift+f', (e) => {
           e.preventDefault();
           if (focusedArea !== 'editor') {
             openSearch();
           }
         });
         
         // Ctrl+F is reserved for Monaco when editor is focused
         useHotkeys('ctrl+f', (e) => {
           if (focusedArea === 'canvas') {
             e.preventDefault();
             openSearch();
           }
           // If editor is focused, do nothing - let Monaco handle it
         });
       }
       ```
    
    Per D-14, D-15, D-16, D-11, D-12. Shortcuts respect context and Monaco's native behavior.
